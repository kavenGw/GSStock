```python
# scripts/_luoboteke_data.py
import sys, json, urllib.request
from pathlib import Path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

out = {}

# 1) 实时价（腾讯 HTTP，GBK，~分隔）
req = urllib.request.Request("http://qt.gtimg.cn/q=sz300757",
                             headers={"User-Agent": "Mozilla/5.0"})
raw = urllib.request.urlopen(req, timeout=10).read().decode("gbk", "ignore")
f = raw.split("~")
out["realtime"] = {
    "name": f[1], "price": f[3], "prev_close": f[4], "open": f[5],
    "change_pct": f[32], "PE_TTM": f[39],
    "year_high": f[41] if len(f) > 41 else None,
    "year_low": f[42] if len(f) > 42 else None,
    "mktcap_yi": f[45], "PB": f[46],
}

# 2) 多年财务时序
import akshare as ak
try:
    df = ak.stock_financial_abstract_ths(symbol="300757", indicator="按年度")
    out["financial_abstract"] = df.to_dict(orient="records")
except Exception as e:
    out["financial_abstract_error"] = f"{type(e).__name__}: {e}"

# 3) PE/PB 5y 分位
for ind, key in [("市盈率(TTM)", "pe_baidu"), ("市净率", "pb_baidu")]:
    try:
        v = ak.stock_zh_valuation_baidu(symbol="300757", indicator=ind, period="近5年")
        out[key] = v.tail(3).to_dict(orient="records")
        out[key + "_describe"] = v[v.columns[1]].describe().to_dict()
    except Exception as e:
        out[key + "_error"] = f"{type(e).__name__}: {e}"

# 4) 主营构成（最新报告期，定 sector/海外敞口/三段 SOTP 基数）
try:
    z = ak.stock_zygc_em(symbol="SZ300757")
    latest = z[z["报告日期"] == z["报告日期"].max()]
    out["zygc_latest"] = latest.to_dict(orient="records")
except Exception as e:
    out["zygc_error"] = f"{type(e).__name__}: {e}"

dst = Path(__file__).resolve().parent.parent / ".omc" / "artifacts"
dst.mkdir(parents=True, exist_ok=True)
(dst / "luoboteke_data.json").write_text(
    json.dumps(out, ensure_ascii=False, indent=2, default=str), encoding="utf-8")
print("OK realtime:", out["realtime"]["name"], out["realtime"]["price"],
      "mktcap_yi=", out["realtime"]["mktcap_yi"], "PB=", out["realtime"]["PB"])
```
